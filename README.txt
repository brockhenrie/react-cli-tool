Creates react components in current directory. -d for dry run and -t for .tsx
-h for help

to install


Add binary to your path

bash ex:
    export PATH=$PATH:/path/to/install/directory

bash add to home/bin
    mkdir ~/bin
    mkdir ~bin/react
    export PATH=$PATH:~/bin/react/react

windows ex:
    set PATH=%PATH%;C:/path/to/install/directory

set go env
    bash ex:
        go env -w GOBIN=/path/to/your/bin

    windows ex:
        go env -w GOBIN=C:\path\to\your\bin

go install
